public static boolean setWay(int[][] map, int i, int j) {
    if(map[map.length - 2][map[0].length - 2] == 2) {
        return true;
    } else {
        if(map[i][j] == 0) {
            map[i][j] = 2;
            if(setWay(map, i+1, j)) {
                return true;
            } else if(setWay(map, i, j+1)) {
                return true;
            } else if(setWay(map, i-1, j)) {
                return true;
            } else if(setWay(map, i, j-1)) {
                return true;
            } else {
                map[i][j] = 3;
                return false;
            }
        } else {
            return false;
        }
    }
}
